rl.on("line", function(line) {

	var n = Number(line.split(' ')[0]);
	var m = Number(line.split(' ')[1]);
	var ownCoupon = n * m;
	var pizza = 0;
	
	while(ownCoupon > m) {
		var newPizza1 = 0;
		newPizza = Math.floor(ownCoupon/m);
		newPizza1 += newPizza;
		ownCoupon = ownCoupon % m;
		ownCoupon += newPizza;
		pizza += newPizza;
	}
	console.log(pizza);
	
	rl.close();
}).on("close", function() {
	process.exit();
});
